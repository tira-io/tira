 35def register_dataset_from_re_rank_file(ir_dataset_id, df_re_rank, original_ir_datasets_id=None):
 36    """
 37    Load a dynamic ir_datasets integration from a given re_rank_file.
 38    The dataset will be registered for the id ir_dataset_id.
 39    The original_ir_datasets_id is used to infer the class of documents, qrels, and queries.
 40    """
 41    import ir_datasets
 42    from ir_datasets.datasets.base import Dataset
 43    original_dataset = ir_datasets.load(original_ir_datasets_id) if original_ir_datasets_id else None
 44
 45    docs = __docs(df_re_rank, original_dataset, True)
 46    queries = __queries(df_re_rank, original_dataset)
 47    qrels = __lazy_qrels(None, original_dataset)
 48    scoreddocs = __scored_docs(df_re_rank, original_dataset)
 49
 50    dataset = Dataset(docs, queries, qrels, scoreddocs)
 51    dataset.metadata = None
 52    ir_datasets.registry._allow_overwrite = True
 53    ir_datasets.registry.register(ir_dataset_id, dataset)
 54
 55    __check_registration_was_successful(ir_dataset_id, original_ir_datasets_id is None)

 67def __docs(input_file, original_dataset, load_default_text):
 68    from ir_datasets.formats import BaseDocs, GenericDoc
 69
 70    class DynamicDocs(BaseDocs):
 71        def __init__(self, input_file, load_default_text):
 72            self.docs = None
 73            self.input_file = input_file
 74            self.load_default_text = load_default_text
 75
 76        def docs_iter(self):
 77            return self.stream_docs()
 78
 79        def docs_count(self):
 80            return sum(1 for _ in self.stream_docs())
 81
 82        def docs_store(self):
 83            return self.__parsed_docs()
 84
 85        def __parsed_docs(self):
 86            if self.docs is None:
 87                d = {}
 88                for i in self.stream_docs():
 89                    d[i.doc_id] = i
 90
 91                self.docs = d
 92
 93            return self.docs
 94
 95        def stream_docs(self):
 96            already_covered = set()
 97            for i in stream_all_lines(self.get_input_file(), self.load_default_text):
 98                if i['docno'] not in already_covered:
 99                    already_covered.add(i['docno'])
100                    yield GenericDoc(doc_id=i['docno'], text=i['text'])
101
102        def get_input_file(self):
103            if type(self.input_file) is str:
104                return self.input_file
105
106            ret = self.input_file()
107            if os.path.isfile(ret + '/documents.jsonl.gz'):
108                return ret + '/documents.jsonl.gz'
109            else:
110                return ret + '/documents.jsonl'
111
112    return DynamicDocs(input_file, load_default_text)

115def __lazy_qrels(input_file, original_qrels):
116    from ir_datasets.formats import BaseQrels, TrecQrel, TrecQrels
117    from ir_datasets.util.download import LocalDownload
118
119    if input_file is None and original_qrels is None:
120        return
121
122    class QrelsFromTira(BaseQrels):
123        def __init__(self, input_file, original_qrels):
124            self.qrels = None if not original_qrels else original_qrels
125            self.input_file = input_file
126            self.original_qrels = original_qrels
127
128        def qrels_cls(self):
129            return TrecQrel if not self.original_qrels else self.original_qrels.qrels_cls()
130
131        def qrels_iter(self):
132            if not self.qrels:
133                qrels_file = self.input_file() + '/qrels.txt'
134
135                self.qrels = TrecQrels(LocalDownload(qrels_file), {})

137            return self.qrels.qrels_iter()

139        def qrels_defs(self):
140            self.qrels_iter()
141            return self.qrels.qrels_defs()
142
143    return QrelsFromTira(input_file, original_qrels)

146def __queries(input_file, original_dataset):
147    from ir_datasets.formats import BaseQueries
148
149    class DynamicQueries(BaseQueries):
150        def __init__(self, input_file):
151            self.queries = None
152            self.input_file = input_file
153
154        def queries_iter(self):
155            return deepcopy(self.__get_queries()).values().__iter__()
156
157        def queries_cls(self):
158            return TirexQuery
159
160        def get_input_file(self):
161            if type(self.input_file) is str:
162                return self.input_file
163
164            return self.input_file() + '/queries.jsonl'
165
166        def __get_queries(self):
167            if self.queries is None:
168                ret = {}
169                for i in stream_all_lines(self.get_input_file(), False):
170                    orig_query = None if 'original_query' not in i else i['original_query']
171                    description = None if (not orig_query or type(orig_query) is not dict or 'description' not in orig_query) else orig_query['description']
172                    narrative = None if (not orig_query or type(orig_query) is not dict or 'narrative' not in orig_query) else orig_query['narrative']
173                    if i['qid'] not in ret:
174                        ret[i['qid']] = TirexQuery(query_id=i['qid'],  text=i['query'], query=i['query'], title=i['query'], description=description, narrative=narrative)
175
176                self.queries = ret
177
178            return self.queries
179
180    return DynamicQueries(input_file)

183def __scored_docs(input_file, original_dataset):
184    from ir_datasets.formats import BaseScoredDocs
185
186    class GenericScoredDocWithRank(NamedTuple):
187        query_id: str
188        doc_id: str
189        score: float
190        rank: int
191
192    class DynamicScoredDocs(BaseScoredDocs):
193        def __init__(self, docs):
194            self.docs = docs
195
196        def scoreddocs_iter(self):
197            return deepcopy(self.docs).__iter__()
198
199        def scoreddocs_count(self) -> int:
200            return len(docs)
201
202    docs = []
203
204    for i in stream_all_lines(input_file, True):
205        docs += [GenericScoredDocWithRank(i['qid'], i['docno'], i['score'], i['rank'])]
206
207    return DynamicScoredDocs(docs)

235def ir_dataset_from_tira_fallback_to_original_ir_datasets():
236    from ir_datasets.datasets.base import Dataset
237
238    def get_download_dir_from_tira(tira_path, truth_dataset) -> "TiraClient":
239        if len(tira_path.split('/')) != 2:
240            logging.info(f'Please pass tira_path as <task>/<tira-dataset>. Got {tira_path}')
241            raise ValueError(f'Please pass tira_path as <task>/<tira-dataset>. Got {tira_path}')
242
243        from tira.rest_api_client import Client as RestClient
244        task, dataset = tira_path.split('/')
245        return RestClient().download_dataset(task, dataset, truth_dataset=truth_dataset)
246
247    class IrDatasetsFromTira():
248        def load(self, dataset_id):
249            try:
250                return original_ir_datasets_load(dataset_id)
251            except Exception:
252                logging.info(f'Load ir_dataset "{dataset_id}" from tira.')
253                docs = self.lazy_docs(lambda: get_download_dir_from_tira(dataset_id, False), None, True)
254                queries = self.lazy_queries(lambda: get_download_dir_from_tira(dataset_id, True), None)
255                qrels = self.lazy_qrels(lambda: get_download_dir_from_tira(dataset_id, True), None)
256                return Dataset(docs, queries, qrels)
257
258        def topics_file(self, tira_path):
259            return get_download_dir_from_tira(tira_path, True) + '/queries.xml'
260
261    ret = IrDatasetsFromTira()
262    ret.lazy_docs = __docs
263    ret.lazy_queries = __queries
264    ret.lazy_qrels = __lazy_qrels
265
266    return ret

269def __check_registration_was_successful(ir_dataset_id, ignore_qrels=True):
270    import ir_datasets
271    dataset = ir_datasets.load(ir_dataset_id)
272
273    assert dataset.has_docs(), "dataset has no documents"
274    assert dataset.has_queries(), "dataset has no queries"
275    assert ignore_qrels or dataset.has_qrels(), "dataset has no qrels"
276    assert dataset.has_scoreddocs(), "dataset has no scored_docs"

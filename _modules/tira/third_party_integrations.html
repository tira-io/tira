 31def get_preconfigured_chatnoir_client(config_directory, features=['TARGET_URI'], num_results=10, retries=25, page_size=10):
 32    from chatnoir_pyterrier import ChatNoirRetrieve
 33    from chatnoir_api import Index as ChatNoirIndex
 34    from chatnoir_pyterrier.feature import Feature
 35
 36    chatnoir_config = json.load(open(config_directory + '/chatnoir-credentials.json'))

 72def load_rerank_data(default, load_default_text=True):

 74
 75    if not os.path.isdir(default_input) and len(default.split('/')) == 2:
 76        from tira.rest_api_client import Client as RestClient

 78
 79    if not default_input.endswith('rerank.jsonl') and not default_input.endswith('rerank.jsonl.gz'):
 80        if os.path.isfile(default_input + '/rerank.jsonl.gz'):

125def normalize_run(run, system_name, depth=1000):
126    try:
127        run['qid'] = run['qid'].astype(int)
128    except Exception:
129        pass
130
131    run['system'] = system_name
132    run = run.copy().sort_values(["qid", "score", "docno"], ascending=[True, False, False]).reset_index()
133
134    if 'Q0' not in run.columns:
135        run['Q0'] = 0
136
137    run = run.groupby("qid")[["qid", "Q0", "docno", "score", "system"]].head(depth)
138
139    # Make sure that rank position starts by 1
140    run["rank"] = 1
141    run["rank"] = run.groupby("qid")["rank"].cumsum()
142
143    return run[['qid', 'Q0', 'docno', 'rank', 'score', 'system']]

147    # Detect if we are in the TIRA sandbox
148    if 'TIRA_INPUT_DATASET' in os.environ:

152            import ir_datasets as original_ir_datasets
153            register_rerank_data_to_ir_datasets(os.environ['TIRA_INPUT_DATASET'], 'dynamic-ds-in-tira')

156
157        return static_ir_dataset(os.environ['TIRA_INPUT_DATASET'])
158    else:
159        try:
